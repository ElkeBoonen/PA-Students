
        //"https://learn.microsoft.com/en-us/dotnet/api/system.string.compare?view=net-9.0"
        //Value	Condition
        //Less than zero	The substring in strA precedes the substring in strB in the sort order.
        //Zero	The substrings occur in the same position in the sort order, or length is zero.
        //Greater than zero	The substring in strA follows the substring in strB in the sort order.

        static int CompareStringsLength(string s1, string s2)
        {
            if (s1.Length < s2.Length) return -1;
            else if (s1.Length > s2.Length) return 1;
            return 0;
        }

        static int CompareStringsASCII(string s1, string s2)
        {
            int asscis1 = 0;
            foreach (char c in s1) asscis1 += (int)c;

            int asscis2 = 0;
            foreach (char c in s2) asscis2 += (int)c;

            if (asscis1 < asscis2) return -1;
            else if (asscis1 > asscis2) return 1;
            return 0;
        }

        static int CompareStringsAs(string s1, string s2)
        {
            int countas1 = 0;
            foreach (char c in s1)
            {
                if (c == 'a') countas1++;
            }

            int countas2 = 0;
            foreach (char c in s2) if (c == 'a') countas2++;;

            if (countas1 < countas2) return -1;
            else if (countas1 > countas2) return 1;
            return 0;
        }


        static void Main(string[] args)
        {

            string[] array = new string[] { "banaan", "citroen", "mango", "kiwi", "lappel", "passievrucht", "appel" };
            Console.WriteLine(String.Join(" ", array));

            Array.Sort(array);
            Console.WriteLine(String.Join(" ", array));

            array = new string[] { "banaan", "citroen", "mango", "kiwi", "lappel", "passievrucht", "appel" };
            Array.Sort(array, CompareStringsLength);
            Console.WriteLine(String.Join(" ", array));

            array = new string[] { "banaan", "citroen", "mango", "kiwi", "lappel", "passievrucht", "appel" };
            Array.Sort(array, CompareStringsASCII);
            Console.WriteLine(String.Join(" ", array));

            array = new string[] { "banaan", "citroen", "mango", "kiwi", "lappel", "passievrucht", "appel" };
            Array.Sort(array, CompareStringsAs);
            Console.WriteLine(String.Join(" ", array));
